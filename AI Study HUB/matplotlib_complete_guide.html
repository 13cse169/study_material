What is Matplotlib?

Python's foundational visualization library — 2D plots, charts, and figures

Matplotlib is the most widely used plotting library in Python. It powers visualizations in Jupyter notebooks, research papers, dashboards, and production systems. Almost every Python visualization library (Seaborn, Pandas plots, Plotly) uses Matplotlib under the hood.

Anatomy of a Figure

Understanding Figure, Axes, Axis, and every visual component

Understanding the object hierarchy is critical. Every element in Matplotlib is an object you can access and customize.

ObjectWhat It IsHow to Access
FigureTop-level container — the whole canvasfig = plt.figure()
AxesA single plot area (has x-axis, y-axis, data)fig, ax = plt.subplots()
AxisThe actual x or y axis line with ticksax.xaxis, ax.yaxis
ArtistEverything drawn: lines, text, patchesreturned by plot commands
Line2DA line drawn on an Axesline, = ax.plot(x, y)
PathPatchFilled shapes (bars, polygons)ax.bar(), ax.add_patch()
TextAny label or annotationax.set_title(), ax.text()
Python — exploring the object hierarchy
import matplotlib.pyplot as plt
import numpy as np

# Create figure and axes
fig, ax = plt.subplots(figsize=(8, 5))

# ── FIGURE properties ────────────────────────────────
fig.set_size_inches(10, 6)
fig.set_dpi(120)
fig.patch.set_facecolor('#f8f8f8')  # figure background
print(fig.get_axes())                # list of all Axes in figure

# ── AXES properties ──────────────────────────────────
ax.set_facecolor('white')           # plot area background
ax.set_aspect('equal')             # force equal x/y scaling
print(type(ax))                      # matplotlib.axes._axes.Axes

# ── AXES COMPONENTS ──────────────────────────────────
ax.xaxis                             # XAxis object
ax.yaxis                             # YAxis object
ax.spines['top']                    # Spine (border) objects
ax.spines['right']
ax.spines['left']
ax.spines['bottom']
ax.title                             # Text object for title
ax.xaxis.label                       # Text object for x-label
ax.yaxis.label                       # Text object for y-label

# ── COMMON AXIS SETUP ────────────────────────────────
x = np.linspace(0, 2*np.pi, 200)
ax.plot(x, np.sin(x))

ax.set_title('Sine Wave', fontsize=16, fontweight='bold', pad=12)
ax.set_xlabel('Angle (radians)', fontsize=13)
ax.set_ylabel('Amplitude', fontsize=13)
ax.set_xlim(0, 2*np.pi)
ax.set_ylim(-1.2, 1.2)

# ── SPINES ───────────────────────────────────────────
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.spines['left'].set_color('#ccc')
ax.spines['bottom'].set_linewidth(1.5)
ax.spines['bottom'].set_color('#333')

plt.tight_layout()
plt.show()

The pyplot Interface

State-machine API — fast, convenient, great for quick exploratory plots

Python — pyplot basics and figure management
import matplotlib.pyplot as plt
import numpy as np

# ── FIGURE CREATION ──────────────────────────────────
plt.figure()                         # default 6.4 × 4.8 inches
plt.figure(figsize=(10, 6))          # width × height in inches
plt.figure(figsize=(8, 5), dpi=150) # high-res
plt.figure(num=1)                    # named/numbered figure
plt.figure(facecolor='#f0f0f0')     # background color

# ── PYPLOT STATE MACHINE ─────────────────────────────
# plt functions always operate on the CURRENT figure/axes
plt.figure(figsize=(8, 4))
x = np.linspace(0, 10, 200)

plt.plot(x, np.sin(x))              # plot to current axes
plt.plot(x, np.cos(x))              # add second line (same axes)
plt.title('Trig Functions')
plt.xlabel('x')
plt.ylabel('y')
plt.legend(['sin(x)', 'cos(x)'])
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.show()

# ── FIGURE MANAGEMENT ────────────────────────────────
plt.clf()                            # clear current figure (keeps window)
plt.cla()                            # clear current axes only
plt.close()                          # close current figure
plt.close('all')                     # close ALL open figures
plt.close(2)                         # close figure number 2

plt.gcf()                            # get current figure
plt.gca()                            # get current axes

# ── SUBPLOT SHORTHAND ────────────────────────────────
plt.subplot(2, 2, 1)                # 2 rows, 2 cols, plot #1
plt.plot([1, 2, 3], [4, 5, 6])
plt.subplot(2, 2, 2)                # switch to plot #2
plt.plot([1, 2, 3], [6, 5, 4])
plt.subplot(223)                     # shorthand: 2 rows,2 cols,#3
plt.subplot(224)
plt.tight_layout()
plt.show()
💡
pyplot vs. OO API: Use plt.* for quick one-off plots. Use the Object-Oriented API (fig, ax = plt.subplots()) for anything you'll reuse, style precisely, or embed in a function. See topic 12 for the OO API.

Line Plots

The most fundamental chart — plotting continuous data against an axis

Python — Line plots, every parameter
import matplotlib.pyplot as plt
import numpy as np

fig, axes = plt.subplots(1, 3, figsize=(15, 4))
x = np.linspace(0, 2*np.pi, 200)

# ── BASIC LINE ───────────────────────────────────────
axes[0].plot(x, np.sin(x))          # simplest form
axes[0].set_title('Basic')

# ── STYLED LINE ──────────────────────────────────────
axes[1].plot(x, np.sin(x),
    color='#4e9eff',               # hex color
    linewidth=2.5,                  # lw= alias
    linestyle='--',                 # ls= alias: '-','--','-.',':'
    alpha=0.9,                      # transparency
    label='sin(x)'
)
axes[1].plot(x, np.cos(x),
    color='#ff7b2f', lw=2, ls='-.',
    label='cos(x)'
)
axes[1].legend()
axes[1].set_title('Styled')

# ── LINE WITH MARKERS ────────────────────────────────
x_few = np.linspace(0, 2*np.pi, 20)
axes[2].plot(x_few, np.sin(x_few),
    marker='o',                     # marker shape
    markersize=8,                    # ms= alias
    markerfacecolor='white',        # hollow markers
    markeredgecolor='#4e9eff',
    markeredgewidth=2,
    linewidth=1.5, color='#4e9eff'
)
axes[2].set_title('With Markers')

plt.tight_layout()
plt.show()

# ── ALL LINESTYLE OPTIONS ────────────────────────────
fig, ax = plt.subplots(figsize=(8, 4))
styles = ['-', '--', '-.', ':']
names  = ['solid', 'dashed', 'dashdot', 'dotted']
for i, (s, n) in enumerate(zip(styles, names)):
    ax.plot(x, np.sin(x) + i, ls=s, lw=2, label=n)
ax.legend(loc='upper right')

# ── ALL MARKER OPTIONS ────────────────────────────────
markers = ['.',',','o','v','^','<','>','s','p','*','h','+','x','D','d','|','_']
# point   pixel   circle  triangle_down  triangle_up  ...

# ── FILL BETWEEN LINES ───────────────────────────────
fig, ax = plt.subplots()
x = np.linspace(0, 10, 200)
y1 = np.sin(x)
y2 = np.sin(x) + 0.5
ax.plot(x, y1, color='steelblue')
ax.plot(x, y2, color='tomato')
ax.fill_between(x, y1, y2,
    alpha=0.3, color='steelblue',  # fill between two lines
    where=(y2 > y1)                  # only fill where condition
)

# ── VERTICAL / HORIZONTAL LINES ──────────────────────
ax.axhline(y=0, color='black', lw=0.8, ls='--')    # horizontal ref line
ax.axvline(x=5, color='red', lw=1, ls=':')           # vertical ref line
ax.axhspan(0.5, 1.0, alpha=0.1, color='yellow')     # horizontal band
ax.axvspan(2, 4, alpha=0.1, color='green')           # vertical band
plt.show()

Scatter Plots

Show relationships between two variables — encode a third via size or color

Python — scatter, bubble charts, color encoding
import matplotlib.pyplot as plt
import numpy as np

np.random.seed(42)
n = 150
x = np.random.randn(n)
y = x * 0.8 + np.random.randn(n) * 0.5
size   = np.random.uniform(20, 300, n)   # for bubble chart
colors = np.random.randn(n)              # for color encoding

fig, axes = plt.subplots(1, 3, figsize=(15, 4))

# ── BASIC SCATTER ────────────────────────────────────
axes[0].scatter(x, y)
axes[0].set_title('Basic')

# ── STYLED SCATTER ───────────────────────────────────
axes[1].scatter(x, y,
    s=60,                             # marker SIZE (points^2)
    c='steelblue',                    # color
    marker='o',                       # marker shape
    alpha=0.7,                        # transparency
    edgecolors='white',               # marker edge color
    linewidths=0.5                    # edge width
)
axes[1].set_title('Styled')

# ── BUBBLE + COLORMAP ────────────────────────────────
sc = axes[2].scatter(x, y,
    s=size,                            # variable size = bubble
    c=colors,                          # variable color
    cmap='coolwarm',                  # colormap
    alpha=0.8,
    edgecolors='none'
)
plt.colorbar(sc, ax=axes[2], label='Value')
axes[2].set_title('Bubble + Colormap')

plt.tight_layout(); plt.show()

# ── CATEGORICAL SCATTER ──────────────────────────────
fig, ax = plt.subplots(figsize=(7, 5))
groups = {'A': (0, 0), 'B': (2, 1), 'C': (-1, 2)}
colors_g = ['#4e9eff', '#ff7b2f', '#3ecf8e']
for (name, (mx, my)), clr in zip(groups.items(), colors_g):
    gx = np.random.randn(50) + mx
    gy = np.random.randn(50) + my
    ax.scatter(gx, gy, c=clr, label=f'Group {name}', alpha=0.7, s=50)
ax.legend()
ax.set_title('Categorical Scatter')
plt.tight_layout(); plt.show()

# ── ADD TREND LINE ────────────────────────────────────
fig, ax = plt.subplots()
ax.scatter(x, y, alpha=0.5, s=40, color='steelblue')
# Linear regression line
m, b = np.polyfit(x, y, 1)          # slope, intercept
x_line = np.linspace(x.min(), x.max(), 100)
ax.plot(x_line, m*x_line + b, color='tomato', lw=2, label=f'y={m:.2f}x+{b:.2f}')
ax.legend(); plt.show()

Bar Charts

Grouped, stacked, horizontal, and error bars

Python — All bar chart variants
import matplotlib.pyplot as plt
import numpy as np

categories = ['Q1', 'Q2', 'Q3', 'Q4']
sales_a    = [42, 58, 71, 63]
sales_b    = [38, 45, 55, 80]
x = np.arange(len(categories))
w = 0.35                               # bar width

fig, axes = plt.subplots(2, 2, figsize=(12, 8))

# ── SIMPLE BAR ───────────────────────────────────────
axes[0,0].bar(categories, sales_a,
    color='steelblue',               # fill color
    edgecolor='navy',                # border color
    linewidth=0.8,
    width=0.6,                        # bar width (0-1)
    alpha=0.85
)
axes[0,0].set_title('Simple Bar')

# ── GROUPED BAR ──────────────────────────────────────
b1 = axes[0,1].bar(x - w/2, sales_a, w, label='Product A', color='#4e9eff')
b2 = axes[0,1].bar(x + w/2, sales_b, w, label='Product B', color='#ff7b2f')
axes[0,1].set_xticks(x); axes[0,1].set_xticklabels(categories)
axes[0,1].legend()
axes[0,1].set_title('Grouped Bar')
# Add value labels on bars
axes[0,1].bar_label(b1, padding=3, fontsize=9)
axes[0,1].bar_label(b2, padding=3, fontsize=9)

# ── STACKED BAR ──────────────────────────────────────
axes[1,0].bar(categories, sales_a, label='Product A', color='#4e9eff')
axes[1,0].bar(categories, sales_b, bottom=sales_a,  # stack on top
    label='Product B', color='#ff7b2f')
axes[1,0].legend()
axes[1,0].set_title('Stacked Bar')

# ── HORIZONTAL BAR WITH ERROR ────────────────────────
errors = [3, 5, 4, 7]
axes[1,1].barh(categories, sales_a,
    xerr=errors,                       # error bars
    capsize=5,                         # error bar cap width
    color='#3ecf8e',
    error_kw={'ecolor':'gray', 'alpha':0.8}
)
axes[1,1].set_title('Horizontal + Error Bars')

plt.tight_layout(); plt.show()

# ── COLORFUL BARS ─────────────────────────────────────
fig, ax = plt.subplots(figsize=(8, 4))
vals   = [42, 58, 71, 63, 55, 49]
labels = ['Jan','Feb','Mar','Apr','May','Jun']
clrs = ['#4e9eff' if v >= max(vals) else '#6080aa' for v in vals]
bars = ax.bar(labels, vals, color=clrs, edgecolor='none', width=0.6)
ax.bar_label(bars, padding=4)
ax.set_ylim(0, max(vals) * 1.2)
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
plt.show()

Histograms & Distributions

Visualize the distribution of numerical data

Python — Histograms, KDE, box plots, violin
import matplotlib.pyplot as plt
import numpy as np

np.random.seed(0)
data = np.random.normal(50, 15, 1000)

fig, axes = plt.subplots(2, 3, figsize=(15, 8))

# ── BASIC HISTOGRAM ──────────────────────────────────
axes[0,0].hist(data,
    bins=30,                          # number of bins
    color='steelblue',
    edgecolor='white',
    linewidth=0.5
)
axes[0,0].set_title('Basic Histogram')
axes[0,0].set_xlabel('Value')
axes[0,0].set_ylabel('Count')

# ── DENSITY (NORMED) ─────────────────────────────────
axes[0,1].hist(data, bins=30,
    density=True,                     # normalize to probability density
    color='#4e9eff', alpha=0.7,
    edgecolor='none'
)
# Overlay KDE manually
from scipy.stats import gaussian_kde
kde = gaussian_kde(data)
x_range = np.linspace(data.min(), data.max(), 200)
axes[0,1].plot(x_range, kde(x_range), color='#ff7b2f', lw=2)
axes[0,1].set_title('Density + KDE')

# ── MULTIPLE OVERLAPPING ─────────────────────────────
d1 = np.random.normal(45, 10, 500)
d2 = np.random.normal(60, 12, 500)
axes[0,2].hist(d1, bins=25, alpha=0.6, label='Group A', color='#4e9eff')
axes[0,2].hist(d2, bins=25, alpha=0.6, label='Group B', color='#ff7b2f')
axes[0,2].legend()
axes[0,2].set_title('Overlapping')

# ── BOX PLOT ─────────────────────────────────────────
groups = [np.random.normal(m, 8, 100) for m in [40,50,55,45]]
bp = axes[1,0].boxplot(groups,
    labels=['A','B','C','D'],
    patch_artist=True,                # fill boxes with color
    notch=False,                       # notched boxes show CI
    showfliers=True,                   # show outlier points
    medianprops={'color':'white', 'linewidth':2}
)
colors_bp = ['#4e9eff','#ff7b2f','#3ecf8e','#b57eff']
for patch, clr in zip(bp['boxes'], colors_bp):
    patch.set_facecolor(clr); patch.set_alpha(0.7)
axes[1,0].set_title('Box Plot')

# ── VIOLIN PLOT ──────────────────────────────────────
vp = axes[1,1].violinplot(groups,
    positions=[1,2,3,4],
    showmeans=True,
    showmedians=True,
    showextrema=True
)
for pc in vp['bodies']:
    pc.set_facecolor('#4e9eff'); pc.set_alpha(0.6)
axes[1,1].set_title('Violin Plot')

# ── STEP HISTOGRAM ───────────────────────────────────
axes[1,2].hist(data, bins=30, histtype='step',
    color='tomato', lw=2)
axes[1,2].hist(data, bins=30, histtype='stepfilled',
    color='tomato', alpha=0.2)
axes[1,2].set_title('Step Histogram')

plt.tight_layout(); plt.show()

Labels, Titles & Legends

Make your chart self-explanatory with great text and legends

Python — All text and legend options
import matplotlib.pyplot as plt
import numpy as np

fig, ax = plt.subplots(figsize=(9, 5))
x = np.linspace(0, 10, 200)
ax.plot(x, np.sin(x), label='sin(x)')
ax.plot(x, np.cos(x), label='cos(x)')

# ── TITLES ───────────────────────────────────────────
ax.set_title('Trig Functions',
    fontsize=18,
    fontweight='bold',
    color='#222',
    pad=14,                           # space between title and axes
    loc='left'                        # 'left','center','right'
)
# Subtitle via text (no native subtitle)
ax.text(0.0, 1.01, 'From 0 to 10 radians',
    transform=ax.transAxes,
    fontsize=11, color='gray')

# ── AXIS LABELS ──────────────────────────────────────
ax.set_xlabel('x (radians)', fontsize=13, labelpad=10)
ax.set_ylabel('Amplitude',   fontsize=13, labelpad=10)

# ── TICK LABELS ──────────────────────────────────────
ax.set_xticks([0, np.pi, 2*np.pi, 3*np.pi])
ax.set_xticklabels(['0', 'π', '2π', '3π'], fontsize=12)
ax.tick_params(axis='x',
    labelsize=11,
    rotation=0,
    length=5,                          # tick length
    width=1.5,                         # tick width
    color='#555',                      # tick color
    labelcolor='#333'                  # tick label color
)
# Rotate x-labels (useful for dates/long names)
plt.setp(ax.get_xticklabels(), rotation=45, ha='right')

# ── LEGEND ───────────────────────────────────────────
leg = ax.legend(
    loc='upper right',               # 'best','upper left', (0.8,0.8) etc.
    fontsize=11,
    title='Functions',               # legend title
    title_fontsize=12,
    framealpha=0.9,                   # legend box transparency
    edgecolor='#ccc',                 # legend box border
    fancybox=True,                    # rounded corners
    shadow=False,
    ncols=2,                          # columns in legend
    markerscale=1.5,                  # scale legend markers
    handlelength=1.5,                 # length of legend line
    borderpad=0.8,                    # inner padding
    labelspacing=0.5                  # space between entries
)

# Legend outside plot
ax.legend(loc='upper left', bbox_to_anchor=(1, 1))
plt.tight_layout()
# OR:
fig.legend(loc='lower center',
    bbox_to_anchor=(0.5, -0.05), ncols=3)

# ── GRID ─────────────────────────────────────────────
ax.grid(True,
    which='major',                    # 'major','minor','both'
    axis='both',                      # 'x','y','both'
    linestyle='--',
    linewidth=0.5,
    color='gray',
    alpha=0.4,
    zorder=0                          # draw behind data
)
ax.minorticks_on()                   # enable minor ticks
ax.grid(True, which='minor', alpha=0.15)

plt.tight_layout(); plt.show()

    dpi=300,                          # resolution (72/100/150/300/600)
    bbox_inches='tight',             # trim whitespace — almost always use this
    pad_inches=0.1,                   # padding when bbox_inches='tight'
    facecolor='white',               # background color
    edgecolor='none',                 # edge color
    transparent=False,               # transparent background
    format='png'                      # override file extension

plt.savefig('chart.pdf')             # vector, perfect for LaTeX
plt.savefig('chart.svg')             # vector, web-ready

plt.savefig('chart.jpeg', quality=95)# JPEG (lossy)

buf = io.BytesIO()

buf.seek(0)
png_data = buf.read()                 # bytes — send over HTTP

import matplotlib.pyplot as plt
import matplotlib.ticker as ticker

fig, ax = plt.subplots(figsize=(9, 5))
x = np.linspace(0, 4*np.pi, 300)

line1, = ax.plot(x, np.sin(x), lw=2, color='#4e9eff', label='sin')
line2, = ax.plot(x, np.cos(x), lw=2, color='#ff7b2f', label='cos')

ax.set_title('Trigonometry', fontsize=16)
ax.set_xlabel('Angle')
ax.set_ylabel('Value')
ax.legend()
fig.tight_layout()
plt.show()

# ── USE IN FUNCTION (reusable) ───────────────────────
def make_chart(x, y, title='', color='steelblue'):
    """Returns fig, ax — caller decides where to show/save."""
    fig, ax = plt.subplots(figsize=(8, 4))
    ax.plot(x, y, color=color, lw=2)
    ax.set_title(title)
    ax.spines[['top','right']].set_visible(False)
    ax.grid(True, alpha=0.3)
    return fig, ax

fig, ax = make_chart(x, np.sin(x), 'Sine', 'tomato')
fig.savefig('sine.png', dpi=150, bbox_inches='tight')
plt.close(fig)

# ── TWIN AXES — two y-scales on one plot ─────────────

ax2.set_ylabel('Growth (%)', color=color2)
ax2.tick_params(axis='y', labelcolor=color2)
ax2.axhline(0, color='gray', lw=0.8, ls='--')

# Combined legend
lines1, labels1 = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax1.legend(lines1 + lines2, labels1 + labels2, loc='upper left')

import matplotlib.pyplot   as plt
import matplotlib.gridspec as gridspec
import numpy as np

# ── GRIDSPEC — full control ───────────────────────────
fig = plt.figure(figsize=(12, 8))
gs = gridspec.GridSpec(
    nrows=3, ncols=3,
    figure=fig,
    width_ratios=[2, 1, 1],           # column widths
    height_ratios=[1, 2, 1],          # row heights
    hspace=0.4, wspace=0.3
)

# Create axes with spans
ax1 = fig.add_subplot(gs[0, :])        # row 0, all columns (wide)
ax2 = fig.add_subplot(gs[1, 0])        # row 1, col 0
ax3 = fig.add_subplot(gs[1, 1:3])     # row 1, cols 1-2
ax4 = fig.add_subplot(gs[2, 0:2])     # row 2, cols 0-1
ax5 = fig.add_subplot(gs[2, 2])        # row 2, col 2

x = np.linspace(0, 10, 200)
ax1.plot(x, np.sin(x));  ax1.set_title('Wide top panel')
ax2.scatter(np.random.randn(50), np.random.randn(50), s=20)
ax3.bar(['A','B','C'],[5,8,3])
ax4.plot(x, np.cos(x))
ax5.hist(np.random.randn(100), bins=15)
plt.suptitle('GridSpec Layout', fontsize=16)
plt.tight_layout(); plt.show()

# ── NESTED GRIDSPEC ──────────────────────────────────
fig = plt.figure(figsize=(12, 6))
outer = gridspec.GridSpec(1, 2)          # 2 columns outer

# Left column: single axis
ax_left = fig.add_subplot(outer[0])
ax_left.plot(x, np.sin(x))

# Right column: nested 2×1
inner = gridspec.GridSpecFromSubplotSpec(2, 1,
    subplot_spec=outer[1], hspace=0.4)
ax_r1 = fig.add_subplot(inner[0])
ax_r2 = fig.add_subplot(inner[1])
ax_r1.plot(x[:len(x)//2], np.cos(x[:len(x)//2]))
ax_r2.hist(np.random.randn(200), bins=20)
plt.tight_layout(); plt.show()

fig, axd = plt.subplot_mosaic("""
    AAB
    CDB
    CEE
""", figsize=(12, 8))
# A spans top-left 2 cols, B spans right col, etc.

import matplotlib        as mpl
import matplotlib.pyplot as plt
import numpy  as np
import pandas as pd
import io, base64

# ── REUSABLE CHART BUILDER ───────────────────────────
def line_chart(df: pd.DataFrame,
               x_col: str,
               y_cols: list,
               title: str = '',
               figsize=(10, 5),
               colors: list = None,
               save_path: str = None) -> plt.Figure:
    """
    Build a styled multi-line chart. Returns Figure.
    """
    fig, ax = plt.subplots(figsize=figsize)

    default_colors = ['#4e9eff','#ff7b2f','#3ecf8e','#b57eff','#ff5c6a']
    clrs = colors or default_colors

    for col, clr in zip(y_cols, clrs):
        ax.plot(df[x_col], df[col],
            label=col, color=clr, lw=2, marker='o', ms=4)

    ax.set_title(title, fontsize=15, fontweight='bold', loc='left')
    ax.spines[['top','right']].set_visible(False)
    ax.grid(True, axis='y', alpha=0.3)
    ax.legend(frameon=False)
    fig.tight_layout()

    if save_path:
        fig.savefig(save_path, dpi=150, bbox_inches='tight')

    return fig


# ── FIGURE TO BASE64 (for web embedding) ─────────────
def fig_to_base64(fig: plt.Figure, dpi=120) -> str:
    """Convert figure to base64 PNG string for HTML embedding."""
    buf = io.BytesIO()
    fig.savefig(buf, format='png', dpi=dpi, bbox_inches='tight')
    buf.seek(0)
    b64 = base64.b64encode(buf.read()).decode('utf-8')
    plt.close(fig)
    return f'data:image/png;base64,{b64}'
