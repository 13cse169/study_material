What is Pandas?

The essential Python library for data manipulation, cleaning, and analysis

Pandas provides two core data structures — Series (1D labeled array) and DataFrame (2D labeled table) — plus an enormous toolkit for loading, cleaning, transforming, and analyzing data. Built on NumPy, it is the backbone of nearly all data science work in Python.

Series — 1D Labeled Array

A one-dimensional array with an index — the building block of Pandas

Python — Series creation, indexing, and methods
import pandas as pd
import numpy  as np

# ── CREATION ─────────────────────────────────────────
s1 = pd.Series([10, 20, 30, 40])                       # default int index
s2 = pd.Series([10, 20, 30], index=['a','b','c'])       # custom label index
s3 = pd.Series({'NYC':8.3, 'LA':3.9, 'Chicago':2.7})  # from dict
s4 = pd.Series(np.random.randn(5), name='returns')      # with name
s5 = pd.Series(5, index=range(4))                         # scalar broadcast → [5,5,5,5]

# ── ATTRIBUTES ───────────────────────────────────────
s2.index        # Index(['a','b','c'], dtype='object')
s2.values       # array([10, 20, 30]) — underlying numpy array
s2.dtype        # dtype('int64')
s2.name         # None (unless set)
s2.shape        # (3,)
s2.size         # 3
len(s2)         # 3

# ── INDEXING ─────────────────────────────────────────
s2['a']          # 10  — label access
s2[['a','c']]    # Series with a=10, c=30
s2.loc['b']     # 20  — explicit label
s2.iloc[1]      # 20  — explicit position
s2['a':'b']     # label slice — INCLUSIVE of 'b'!
s2[s2 > 15]     # boolean mask: b=20, c=30

# ── ARITHMETIC — aligns on index ─────────────────────
a = pd.Series([1,2,3], index=['x','y','z'])
b = pd.Series([10,20], index=['x','z'])
a + b
# x    11.0   ← aligned by label
# y     NaN   ← 'y' not in b → NaN
# z    23.0   dtype: float64

# ── USEFUL METHODS ───────────────────────────────────
s3.sort_values()             # sort by value (ascending)
s3.sort_values(ascending=False)
s3.sort_index()              # sort by index label
s3.rank()                    # rank each element
s3.value_counts()            # frequency count of each unique value
s3.value_counts(normalize=True)  # proportions (sum to 1.0)
s3.describe()                # count, mean, std, min, Q1, med, Q3, max
s3.isnull()                  # boolean mask — True where NaN
s3.notnull()                 # opposite
s3.fillna(0)                # replace NaN with 0
s3.dropna()                  # remove NaN values
s3.between(2, 5)             # True if 2 ≤ x ≤ 5
s3.clip(lower=3, upper=8)  # clamp values to [3, 8]
s3.diff()                    # x[t] - x[t-1]
s3.pct_change()              # (x[t] - x[t-1]) / x[t-1]
s3.cumsum()                  # cumulative sum
s3.cumprod()                 # cumulative product
s3.expanding().mean()        # expanding window mean
s3.rolling(3).mean()         # 3-period rolling mean
s3.map(lambda x: x*2)       # apply function element-wise
s3.apply(np.log)             # apply numpy ufunc
s3.rename('pop_millions')   # set name
s3.to_frame()                # convert Series to 1-col DataFrame
s3.reset_index()             # reset index to 0,1,2...

DataFrame — The Core Data Structure

2D labeled table — like a spreadsheet or SQL table in Python

Python — Creating, accessing, and modifying DataFrames
import pandas as pd
import numpy  as np

# ── CREATION ─────────────────────────────────────────
# From dict of lists (most common)
df = pd.DataFrame({
    'name':   ['Alice', 'Bob', 'Charlie', 'Diana'],
    'age':    [28, 34, 22, 31],
    'salary': [75000, 92000, 58000, 88000],
    'dept':   ['Eng', 'Eng', 'HR', 'Finance']
})

# From list of dicts (records)
df2 = pd.DataFrame([
    {'name': 'Alice', 'score': 95},
    {'name': 'Bob',   'score': 87},
])

# From numpy array
df3 = pd.DataFrame(
    np.random.randn(5, 3),
    columns=['A', 'B', 'C'],
    index=['r1','r2','r3','r4','r5']
)

# ── KEY ATTRIBUTES ───────────────────────────────────
df.shape          # (4, 4)  — (rows, columns)
df.columns        # Index(['name','age','salary','dept'])
df.index          # RangeIndex(start=0, stop=4)
df.dtypes         # dtype per column
df.values         # underlying numpy array (object dtype if mixed)
df.size           # 16 total elements
df.ndim           # 2
len(df)           # 4 — number of rows

# ── ACCESSING COLUMNS ────────────────────────────────
df['name']          # Series — single column
df.name             # attribute style (avoid if col name conflicts with methods!)
df[['name','age']]  # DataFrame — multiple columns

# ── ADD / MODIFY COLUMNS ─────────────────────────────
df['bonus']    = df['salary'] * 0.10              # computed column
df['senior']   = df['age'] > 30                   # boolean column
df['full_name'] = df['name'] + ' Smith'           # string column
df.insert(1, 'id', range(1, len(df)+1))           # insert at position 1
df = df.assign(                                     # assign multiple at once
    tax       = df['salary'] * 0.3,
    take_home = lambda d: d['salary'] - d['salary']*0.3
)

# ── DROP COLUMNS / ROWS ──────────────────────────────
df = df.drop(columns=['bonus','full_name'])         # drop columns
df.drop(columns='senior', inplace=True)            # in-place
df = df.drop(index=0)                               # drop row by label
df = df.drop(index=df[df['dept']=='HR'].index)    # conditional drop
df = df.reset_index(drop=True)                     # reset after dropping

# ── RENAME ───────────────────────────────────────────
df.rename(columns={'salary':'pay', 'dept':'department'}, inplace=True)
df.rename(index={0:'first', 1:'second'}, inplace=True)
df.columns = [c.lower().replace(' ','_') for c in df.columns]  # bulk rename

# ── SET INDEX ────────────────────────────────────────
df = df.set_index('name')         # use name column as row labels
df = df.reset_index()              # move index back to column

# ── COPY ─────────────────────────────────────────────
df_copy = df.copy()                # deep copy — changes won't affect original
df_view = df[df['age']>25]        # may be a view or copy — use .copy() to be safe

Python — All I/O formats
import pandas as pd

# ── CSV ───────────────────────────────────────────────
# Read
df = pd.read_csv('data.csv')
df = pd.read_csv('data.csv',
    sep=';',                      # semicolon-delimited
    header=0,                     # row 0 is header (default)
    index_col='id',               # use 'id' column as row index
    usecols=['name','age'],       # only load these columns
    dtype={'age': 'int32'},       # specify dtypes to save memory
    parse_dates=['created_at'],  # auto-parse date column
    na_values=['NA','-','n/a'],  # treat these strings as NaN
    skiprows=2,                   # skip first 2 rows
    nrows=10000,                  # read only first 10k rows
    encoding='utf-8',            # file encoding
    compression='gzip',          # read .gz file directly
    thousands=',',               # parse '1,000' as 1000
    decimal='.'                   # decimal separator
)
# Write
df.to_csv('output.csv', index=False)            # no row numbers
df.to_csv('output.csv', float_format='%.2f')   # 2 decimal places
df.to_csv('output.csv.gz', compression='gzip') # write compressed

# ── EXCEL ─────────────────────────────────────────────

Always explore a new dataset before doing anything else

import pandas as pd

df = pd.read_csv('data.csv')

# ── FIRST LOOK ───────────────────────────────────────
df.head()           # first 5 rows (default)

df.columns.tolist() # list of column names
df.dtypes           # data type per column
df.info()           # dtype + non-null count + memory — ESSENTIAL

df.memory_usage(deep=True).sum() / 1e6  # total MB

# ── STATISTICS ───────────────────────────────────────
df.describe()                        # count/mean/std/min/Q1/Q2/Q3/max (numeric)

df.kurt(numeric_only=True)          # tailedness (0=normal)

df['dept'].value_counts(normalize=True)  # as proportions

def profile(df):
    print(f"Shape:      {df.shape}")
    print(f"Memory:     {df.memory_usage(deep=True).sum()/1e6:.2f} MB")
    missing = df.isnull().sum()
    missing = missing[missing > 0]
    if len(missing):
        print(f"\nMissing:\n{missing}")
    print(f"\nDuplicates: {df.duplicated().sum()}")
    print(f"\nDtypes:\n{df.dtypes.value_counts()}")

profile(df)

Selecting Data — loc, iloc, at, iat

Four indexers — know exactly which one to use and why

Python — All selection methods
import pandas as pd

df = pd.DataFrame({
    'name':   ['Alice','Bob','Charlie','Diana'],
    'age':    [28, 34, 22, 31],
    'salary': [75000, 92000, 58000, 88000],
    'dept':   ['Eng','Eng','HR','Finance']
}, index=['a','b','c','d'])

# ── .loc — LABEL-BASED ───────────────────────────────
df.loc['a']                    # row 'a' as Series
df.loc[['a','c']]              # rows a and c as DataFrame
df.loc['a':'c']               # rows a to c INCLUSIVE (label slicing!)
df.loc['a', 'name']            # single cell → 'Alice'
df.loc['a':'b', 'name':'age'] # rows a-b, cols name to age
df.loc[:, ['name','salary']]   # all rows, specific cols
df.loc[df['age'] > 25]         # boolean filter
df.loc[df['dept']=='Eng', 'salary']  # filter rows + select col

# ── .iloc — INTEGER POSITION ─────────────────────────
df.iloc[0]                     # first row
df.iloc[-1]                    # last row
df.iloc[[0,2]]                 # rows 0 and 2
df.iloc[1:3]                   # rows 1–2 (EXCLUSIVE of 3)
df.iloc[0, 1]                  # cell row=0, col=1 → 28
df.iloc[0:2, 0:2]              # first 2 rows and 2 cols
df.iloc[:, 0]                  # first column as Series
df.iloc[:, -1]                 # last column
df.iloc[::2]                   # every other row
df.iloc[::-1]                  # reversed row order

# ── .at / .iat — SINGLE VALUE (fastest) ─────────────
df.at['a', 'name']             # 'Alice'  — label access
df.iat[0, 0]                   # 'Alice'  — position access
df.at['a', 'salary'] = 80000  # SET single cell
df.iat[0, 2]            = 80000

# ── [] SHORTHAND ─────────────────────────────────────
df['name']                     # column → Series
df[['name','age']]             # columns → DataFrame
df[df['age'] > 25]             # boolean mask → filtered DataFrame

# ── ASSIGNMENT THROUGH INDEXER ───────────────────────
df.loc[df['dept']=='HR', 'salary'] *= 1.1  # 10% raise for HR
df.iloc[0:2, 2] = 90000                    # set range of cells

# ── AVOID CHAINED INDEXING ───────────────────────────
# ❌ df['dept']['a'] = 'Marketing'  → SettingWithCopyWarning
# ✅ df.loc['a', 'dept'] = 'Marketing'  — always use explicit indexer
IndexerRow axisCol axisSingle value?

Filtering Data

Boolean masks, query(), isin(), between() — the Pandas WHERE clause

Python — All filtering techniques
import pandas as pd

df = pd.DataFrame({
    'name':   ['Alice','Bob','Charlie','Diana','Eve'],
    'age':    [28, 34, 22, 31, 27],
    'salary': [75000, 92000, 58000, 88000, 65000],
    'dept':   ['Eng','Eng','HR','Finance','Eng']
})

# ── SINGLE CONDITION ─────────────────────────────────
df[df['age'] > 30]
df[df['dept'] == 'Eng']
df[df['salary'] != 58000]
df[df['name'].str.startswith('A')]

# ── MULTIPLE CONDITIONS — use & | ~ not 'and'/'or'/'not' !
df[(df['age'] > 25) & (df['dept'] == 'Eng')]     # AND
df[(df['salary'] > 80000) | (df['age'] < 25)]   # OR
df[~(df['dept'] == 'HR')]                        # NOT

# ── SPECIAL METHODS ──────────────────────────────────
# isin() — match against a list
df[df['dept'].isin(['Eng','Finance'])]
df[~df['name'].isin(['Bob','Eve'])]           # NOT in list

# between() — inclusive range
df[df['age'].between(25, 32)]                   # 25 ≤ age ≤ 32
df[df['salary'].between(60000, 90000)]

# str accessor for text filtering
df[df['name'].str.contains('li', case=False)]
df[df['dept'].str.match(r'^E')]                 # regex match

# ── query() — cleaner syntax for complex conditions ──
df.query('age > 25 and dept == "Eng"')
df.query('salary > 70000 or age < 25')
df.query('dept in ["Eng", "Finance"]')
df.query('age.between(25, 33)')

min_sal = 70000
df.query('salary > @min_sal')          # reference Python var with @
df.query('index > 2')                   # filter by index value

# ── filter() — filter COLUMNS ────────────────────────
df.filter(items=['name','salary'])
df.filter(like='a')                     # cols whose name contains 'a'
df.filter(regex='^s')                   # cols starting with 's'

# ── nlargest / nsmallest ─────────────────────────────
df.nlargest(3, 'salary')                # top 3 earners
df.nsmallest(2, 'age')                  # 2 youngest
df.nlargest(3, ['salary','age'])       # break ties by age
⚠️
Always use &, |, ~ for combining boolean masks — NOT Python's and, or, not. Wrap each condition in parentheses to avoid operator precedence issues.

Handling Missing Data

Detect, remove, fill, and interpolate NaN values

Python — Complete missing data toolkit
import pandas as pd
import numpy  as np

df = pd.DataFrame({
    'name':   ['Alice','Bob',None,     'Diana'],
    'age':    [28,     np.nan,22,      31],
    'salary': [75000, 92000, np.nan, np.nan],
    'dept':   ['Eng',  'Eng',  None,  'Finance']
})

# ── DETECT ───────────────────────────────────────────
df.isnull()                         # bool DataFrame — True where NaN
df.notnull()                        # opposite
df.isnull().sum()                   # NaN count per column
df.isnull().sum(axis=1)            # NaN count per row
df.isnull().any()                   # True if ANY NaN in column
df.isnull().all()                   # True if ALL NaN in column
df[df['salary'].isnull()]          # rows where salary is NaN

# ── DROP ─────────────────────────────────────────────
df.dropna()                         # drop rows with ANY NaN
df.dropna(axis=1)                  # drop cols with ANY NaN
df.dropna(how='all')               # drop rows where ALL are NaN
df.dropna(thresh=3)                # keep rows with ≥ 3 non-NaN values
df.dropna(subset=['name','age'])  # only check these columns

# ── FILL — scalar ────────────────────────────────────
df['salary'].fillna(0)
df['dept'].fillna('Unknown')
df.fillna({'age':0, 'salary':-1, 'dept':'n/a'})  # per-column values

# ── FILL — statistical ───────────────────────────────
df['salary'].fillna(df['salary'].mean())     # fill with mean
df['salary'].fillna(df['salary'].median())   # median (outlier-robust)
df['dept'].fillna(df['dept'].mode()[0])      # most frequent value

# ── FILL — propagation ───────────────────────────────
df.ffill()                          # forward-fill: copy last valid forward
df.bfill()                          # backward-fill: copy next valid backward
df.ffill(limit=2)                  # fill at most 2 consecutive NaN
df['salary'].ffill().bfill()       # forward then backward (fill all)

# ── INTERPOLATION ────────────────────────────────────
s = pd.Series([1, np.nan, np.nan, 4, np.nan, 10])
s.interpolate()                              # linear (default)
s.interpolate(method='polynomial', order=2) # polynomial fit
s.interpolate(method='spline', order=3)     # cubic spline
s.interpolate(method='nearest')             # nearest neighbour

# ── GROUP-WISE IMPUTATION ────────────────────────────
# Fill salary NaN with the mean salary of the same department
df['salary'] = df.groupby('dept')['salary'].transform(
    lambda x: x.fillna(x.mean())
)

# ── REPLACE SENTINEL VALUES ──────────────────────────
df.replace(-999, np.nan)
df.replace({'age': {-1: np.nan}, 'salary': {0: np.nan}})

Transforming Data

Type conversion, sorting, deduplication, binning, and column operations

Python — Transformation toolkit
import pandas as pd
import numpy  as np

# ── TYPE CONVERSION ──────────────────────────────────
df['age']     = df['age'].astype(int)
df['salary']  = df['salary'].astype(float)
df['active']  = df['active'].astype(bool)
df['dept']    = df['dept'].astype('category')    # saves memory
df['joined']  = pd.to_datetime(df['joined'])
df['revenue'] = pd.to_numeric(df['revenue'], errors='coerce')  # NaN on fail
df = df.convert_dtypes()                            # auto best nullable types (pd 1.0+)

# ── SORTING ──────────────────────────────────────────
df.sort_values('salary')
df.sort_values('salary', ascending=False)
df.sort_values(['dept','salary'], ascending=[True,False])
df.sort_values('salary', na_position='first')
df.sort_index()
df.sort_index(ascending=False)

# ── DUPLICATES ───────────────────────────────────────
df.drop_duplicates()
df.drop_duplicates(subset=['name'])
df.drop_duplicates(subset=['name'], keep='last')  # keep last occurrence
df.drop_duplicates(keep=False)                     # drop ALL copies

# ── COLUMN MATH ──────────────────────────────────────
df['bonus']      = df['salary'] * 0.10
df['tax']        = df['salary'] * 0.30
df['take_home']  = df['salary'] - df['tax']
df['log_salary'] = np.log(df['salary'])

# ── NORMALIZATION ────────────────────────────────────
s = df['salary']
df['salary_minmax'] = (s - s.min()) / (s.max() - s.min())  # [0, 1]
df['salary_zscore'] = (s - s.mean()) / s.std()              # mean=0, std=1
df['salary_rank']   = s.rank(ascending=False, method='dense')

# ── BINNING ──────────────────────────────────────────
df['age_group'] = pd.cut(
    df['age'],
    bins=[0, 25, 35, 50, 100],
    labels=['Young','Adult','Senior','Elder'],
    right=True     # (0,25] — right-closed intervals
)
df['sal_quartile'] = pd.qcut(df['salary'], q=4, labels=['Q1','Q2','Q3','Q4'])

# ── VALUE REPLACEMENT ────────────────────────────────
df['dept'].replace({'Eng':'Engineering', 'HR':'Human Resources'})
df.replace(-999, np.nan)
df.replace(regex={r'^\s+$': np.nan})     # whitespace-only → NaN

# ── CLIPPING ─────────────────────────────────────────
df['salary'].clip(lower=50000, upper=100000)   # cap extreme values

import pandas as pd
import numpy  as np

df = pd.DataFrame({
    'name':   ['Alice','Bob','Charlie'],
    'age':    [28, 34, 22],
    'salary': [75000, 92000, 58000]
})

df['dept'].map({'Eng':1, 'HR':2, 'Finance':3})   # label encode with dict

df.apply(lambda col: col.max() - col.min())          # range per column
df.apply(np.std)                                     # std of each column
df['salary'].apply(lambda x: 'high' if x > 80000 else 'low')

# Apply to each ROW (axis=1) — returns Series per row
df.apply(lambda row: row['salary'] / row['age'], axis=1)

# Return Series from apply → expands to new columns
def comp(row):
    return pd.Series({
        'bonus':     row['salary'] * 0.1,
        'tax':       row['salary'] * 0.3,
        'take_home': row['salary'] * 0.6
    })
df.join(df.apply(comp, axis=1))

# ── DataFrame.map() — element-wise on ALL cells ──────
df.select_dtypes('number').map(lambda x: round(x, 2))

# ── VECTORIZED — ALWAYS prefer when possible! ────────

# ❌ Slow apply
df['salary'].apply(lambda x: x * 1.1)
# ✅ Fast vectorized
df['salary'] * 1.1

# ❌ Slow apply for string
df['name'].apply(lambda x: x.lower())
# ✅ Fast str accessor
df['name'].str.lower()

# ❌ Slow apply for conditionals
df['salary'].apply(lambda x: 'high' if x > 80000 else 'low')
# ✅ Fast np.where
np.where(df['salary'] > 80000, 'high', 'low')

# ❌ Slow apply for multi-choice
df['salary'].apply(lambda x: 'high' if x>80000 else 'mid' if x>65000 else 'low')
# ✅ Fast np.select
np.select(
    [df['salary'] > 80000, df['salary'] > 65000],
    ['high',                'mid'],
    default='low'
)

# ❌ Slow apply for date parsing
df['date'].apply(lambda x: pd.to_datetime(x))
# ✅ Fast
pd.to_datetime(df['date'])

import pandas as pd
import numpy  as np

# ── MEMORY REDUCTION ─────────────────────────────────
def reduce_memory(df):
    """Automatically downcast numeric types and categorize strings."""
    before = df.memory_usage(deep=True).sum() / 1e6
    for col in df.columns:
        ct = df[col].dtype
        if ct == 'object':
            if df[col].nunique() / len(df) < 0.5:  # low cardinality
                df[col] = df[col].astype('category')
        elif np.issubdtype(ct, np.integer):
            df[col] = pd.to_numeric(df[col], downcast='integer')
        elif np.issubdtype(ct, np.floating):
            df[col] = pd.to_numeric(df[col], downcast='float')
    after = df.memory_usage(deep=True).sum() / 1e6
    print(f"Memory: {before:.1f} MB → {after:.1f} MB ({(1-after/before)*100:.0f}% saved)")
    return df

df = reduce_memory(df)

# ── SPECIFY DTYPES ON READ ────────────────────────────
dtypes = {'age': 'int8', 'salary': 'int32', 'dept': 'category'}
df = pd.read_csv('data.csv', dtype=dtypes)  # never loads as object!

# ── CATEGORY TYPE — 10x smaller for low-cardinality ──
df['dept'] = df['dept'].astype('category')
print(df.memory_usage(deep=True))  # compare before/after

# Operations on categoricals are fast (integer-coded internally)
df.groupby('dept')['salary'].mean()  # faster with category dtype

# ── AVOID LOOPS — use vectorized ops ─────────────────
# ❌ 100x slower
results = []
for _, row in df.iterrows():
    results.append(row['salary'] * 1.1)
df['raise'] = results

# ✅ 100x faster
df['raise'] = df['salary'] * 1.1

# ── EVALUATION ENGINE — eval() & query() ─────────────
# For large DataFrames (>10k rows), eval() avoids temp arrays
df.eval('bonus = salary * 0.1', inplace=True)
df.eval('net_salary = salary - salary * 0.3')
df.query('salary > 80000 and dept == "Eng"')  # also uses numexpr

# Process files too large to fit in memory
agg_results = []
for chunk in pd.read_csv('huge.csv', chunksize=100_000):
    agg = chunk.groupby('dept')['salary'].sum()
    agg_results.append(agg)
total = pd.concat(agg_results).groupby(level=0).sum()

# ── PARQUET OVER CSV ─────────────────────────────────
# Parquet: 5-10x smaller, 10x faster to read, preserves dtypes
df.to_parquet('data.parquet')           # write once
df = pd.read_parquet('data.parquet',   # read much faster
                     columns=['name','salary'])  # column pruning!

# ── COPY-ON-WRITE (pandas 2.0+) ──────────────────────

import time

    for _ in range(n): fn()
    print(f"{(time.perf_counter()-t)/n*1000:.2f} ms per call")

ts = pd.Series(np.random.randn(20) * 10 + 100, index=dates, name='price')

# ── ROLLING — fixed-size sliding window ──────────────
ts.rolling(7).mean()            # 7-day moving average
ts.rolling(7).sum()
ts.rolling(7).std()
ts.rolling(7).min()
ts.rolling(7).max()
ts.rolling(7).quantile(0.9)
ts.rolling(7).corr(ts.shift(1))  # rolling autocorrelation

# Minimum periods — avoid NaN at start
ts.rolling(7, min_periods=1).mean()  # use available data at start

# Center the window (for non-causal smoothing)
ts.rolling(7, center=True).mean()   # ±3 days around each point

# Rolling on time offset (not number of rows)
ts.rolling('7D').mean()              # 7 calendar days
ts.rolling('30D').std()              # 30-day volatility

# Multiple columns
df = pd.DataFrame({'a': ts, 'b': ts * 1.05})
df.rolling(5).mean()               # rolling mean of all columns

# Custom window function
ts.rolling(7).apply(lambda x: x[-1]/x[0] - 1)  # 7-day return
ts.rolling(7).apply(np.median, raw=True)        # raw=True for numpy speed

# ── EXPANDING — grows from start ─────────────────────
ts.expanding().mean()              # cumulative mean up to each point
ts.expanding().max()               # running maximum
ts.expanding().std()               # expanding standard deviation
ts.expanding(min_periods=5).mean() # need at least 5 points

# ── EWM — exponentially weighted (recency-biased) ────
ts.ewm(span=7).mean()             # span: controls decay speed
ts.ewm(halflife=3).mean()         # halflife: periods for weight to halve
ts.ewm(alpha=0.3).mean()          # alpha: explicit smoothing factor [0,1]
ts.ewm(com=3).std()               # center-of-mass decay

# ── PRACTICAL: BOLLINGER BANDS ───────────────────────
prices = ts
ma20   = prices.rolling(20).mean()
std20  = prices.rolling(20).std()
upper_band = ma20 + (2 * std20)

import pandas as pd
import numpy  as np

# ── METHOD CHAINING — fluent, readable transforms ────

# ✅ Clean chained version
df = (
    pd.read_csv('raw.csv')
    .rename(columns={'SalAry':'salary'})
    .dropna(subset=['salary'])
    .query('age >= 18')
    .assign(
        salary = lambda d: d['salary'].astype(float),
        bonus  = lambda d: d['salary'] * 0.1
    )
    .sort_values('salary', ascending=False)
    .reset_index(drop=True)
)

# ── .pipe() — call custom functions in chain ─────────
def clean_columns(df):
    """Lowercase and snake_case all column names."""
    df.columns = (df.columns
                  .str.lower()
                  .str.strip()
                  .str.replace(r'\s+', '_', regex=True))
    return df

def remove_outliers(df, col, n_std=3):
    """Remove rows where col is more than n_std from mean."""
    mean, std = df[col].mean(), df[col].std()
    return df[(df[col] - mean).abs() <= n_std * std]

def add_features(df):
    """Engineered features."""
    df = df.assign(
        salary_per_year_exp = df['salary'] / df['years_exp'].clip(lower=1),
        is_senior = df['years_exp'] > 5,
        dept_size = df.groupby('dept')['salary'].transform('count')
    )
    return df

df = (
    pd.read_csv('raw.csv')
    .pipe(clean_columns)
    .pipe(remove_outliers, col='salary', n_std=3)
    .pipe(add_features)
    .query('age >= 18')
    .dropna()
    .reset_index(drop=True)
)

# ── FEATURE ENGINEERING PATTERNS ─────────────────────
def encode_features(df, cat_cols, num_cols):
    # Label encode categoricals
    for col in cat_cols:
        df[col + '_code'] = df[col].astype('category').cat.codes
    # Z-score normalize numerics
    for col in num_cols:
        df[col + '_z'] = (df[col] - df[col].mean()) / df[col].std()
    return df
