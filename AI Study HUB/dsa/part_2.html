# === QUICK SORT — Average O(n log n), In-place ===
def quick_sort(arr, lo=0, hi=None):
    if hi is None: hi = len(arr) - 1
    if lo >= hi: return
    p = partition(arr, lo, hi)
    quick_sort(arr, lo, p-1)
    quick_sort(arr, p+1, hi)

def partition(arr, lo, hi):
    pivot = arr[hi]; i = lo - 1
    for j in range(lo, hi):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    arr[i+1], arr[hi] = arr[hi], arr[i+1]
    return i+1

# === COUNTING SORT — O(n+k) for integers ===
def counting_sort(arr):
    if not arr: return []
    mn, mx = min(arr), max(arr)
    count = [0] * (mx - mn + 1)
    for x in arr: count[x-mn] += 1
    return [x+mn for x, c in enumerate(count) for _ in range(c)]

# === PYTHON SORT — Custom key ===
people = [('Alice', 30), ('Bob', 25), ('Charlie', 35)]
people.sort(key=lambda x: x[1])      # Sort by age
people.sort(key=lambda x: -x[1])     # Descending by age
sorted(words, key=len)               # Sort by string length
TRICK

Dutch National Flag

3-way partition for arrays with 3 distinct values (0,1,2). Three pointers: lo, mid, hi. O(n) time, O(1) space. Classic interview problem.

TRICK

Merge Intervals

Sort by start time, then greedily merge overlapping intervals. O(n log n). Also used in: insert interval, meeting rooms, employee schedules.

TRICK

Kth Largest Element

QuickSelect (avg O(n)), Min-Heap of size k (O(n log k)), or built-in sort. QuickSelect is optimal for single-query kth order statistic.

TOPIC 08

Hashing

HASH MAPS · HASH SETS · COLLISION HANDLING

ℹ️
A hash function maps keys to array indices. The goal: distribute keys uniformly. Python's dict and set are hash tables with O(1) average get/set/delete. Worst case O(n) due to collisions is extremely rare in practice.
STRUCTURE

Hash Map (dict)

Key-value pairs. O(1) average insert, delete, lookup. Handles arbitrary hashable keys. Unordered in Python < 3.7, insertion-ordered in 3.7+.

STRUCTURE

Hash Set (set)

Unique elements. O(1) average add, remove, membership test. Best for deduplication, visited tracking in BFS/DFS, finding complements.

TECHNIQUE

Chaining (Separate)

Each bucket holds a linked list of colliding elements. Simple to implement. Load factor λ = n/m. Performance degrades as λ grows.

TECHNIQUE

Open Addressing

Linear probing, quadratic probing, double hashing. All elements stored in the table itself. Better cache performance than chaining.

# === TWO SUM — Classic hash map usage ===
def two_sum(nums, target):
    seen = {}
    for i, n in enumerate(nums):
        diff = target - n
        if diff in seen: return [seen[diff], i]
        seen[n] = i
    return []

# === SUBARRAY SUM EQUALS K — Prefix sum + hash map ===
def subarray_sum(nums, k):
    prefix = {0: 1}; total = count = 0
    for n in nums:
        total += n
        count += prefix.get(total - k, 0)
        prefix[total] = prefix.get(total, 0) + 1
    return count

# === LRU CACHE — Hash Map + Doubly Linked List ===
from collections import OrderedDict
class LRUCache:
    def __init__(self, cap):
        self.cap = cap
        self.cache = OrderedDict()

    def get(self, key):
        if key not in self.cache: return -1
        self.cache.move_to_end(key)  # mark as recently used
        return self.cache[key]

    def put(self, key, val):
        if key in self.cache: self.cache.move_to_end(key)
        self.cache[key] = val
        if len(self.cache) > self.cap: self.cache.popitem(last=False)

# === ROLLING HASH — Rabin-Karp substring search ===
def rabin_karp(text, pattern):
    n, m = len(text), len(pattern)
    BASE, MOD = 31, 10**9+7
    ph = sum(ord(c)*BASE**i for i,c in enumerate(pattern)) % MOD
    wh = sum(ord(c)*BASE**i for i,c in enumerate(text[:m])) % MOD
    results = [0] if ph==wh else []
    for i in range(1, n-m+1):
        wh = (wh - ord(text[i-1])) * pow(BASE,MOD-2,MOD) % MOD
        if wh == ph: results.append(i)
    return results

# === VALID PARENTHESES ===
def is_valid(s):
    stack = []; pairs = {')':'(', '}':'{', ']':'['}
    for c in s:
        if c in '([{': stack.append(c)
        elif not stack or stack[-1] != pairs[c]: return False
        else: stack.pop()
    return not stack

# === NEXT GREATER ELEMENT — Monotonic Stack ===
def next_greater(nums):
    res = [-1] * len(nums)
    stack = []           # stores indices, decreasing stack
    for i, n in enumerate(nums):
        while stack and nums[stack[-1]] < n:
            res[stack.pop()] = n
        stack.append(i)
    return res

# === LARGEST RECTANGLE IN HISTOGRAM ===
def largest_rect(heights):
    stack = [-1]; max_area = 0
    heights.append(0)   # sentinel
    for i, h in enumerate(heights):
        while stack[-1] != -1 and heights[stack[-1]] >= h:
            height = heights[stack.pop()]
            width = i - stack[-1] - 1
            max_area = max(max_area, height * width)
        stack.append(i)
    return max_area

# === MIN STACK — O(1) getMin ===
class MinStack:
    def __init__(self): self.s = []; self.mins = []
    def push(self, v):
        self.s.append(v)
        self.mins.append(min(v, self.mins[-1] if self.mins else v))
    def pop(self): self.s.pop(); self.mins.pop()
    def get_min(self): return self.mins[-1]

TYPE

Simple Queue

Use collections.deque for O(1) enqueue/dequeue. Never use list.pop(0) — it's O(n). Primary use: BFS traversal of graphs and trees.

TYPE

Deque (Double-Ended)

Push/pop from both ends in O(1). Used in sliding window maximum, palindrome check, and implementing both stack and queue.

TYPE

Priority Queue (Heap)

heapq module — min-heap by default. For max-heap: negate values. Dijkstra's, task scheduling, K closest points, merge K sorted lists.

TYPE

Circular Queue

Fixed-size ring buffer using array + two pointers. O(1) operations. Used in OS scheduling, I/O buffers, streaming data pipelines.

from collections import deque
import heapq

# === BFS — Shortest path in unweighted graph ===
def bfs(graph, start):
    dist = {start: 0}; q = deque([start])
    while q:
        node = q.popleft()
        for nei in graph[node]:
            if nei not in dist:
                dist[nei] = dist[node] + 1
                q.append(nei)
    return dist

# === SLIDING WINDOW MAXIMUM — Monotonic Deque ===
def max_sliding_window(nums, k):
    dq = deque(); res = []
    for i, n in enumerate(nums):
        while dq and nums[dq[-1]] < n: dq.pop()
        dq.append(i)
        if dq[0] == i-k: dq.popleft()   # outside window
        if i >= k-1: res.append(nums[dq[0]])
    return res

# === K CLOSEST POINTS — Priority Queue ===
def k_closest(points, k):
    heap = []
    for x, y in points:
        heapq.heappush(heap, (x**2+y**2, x, y))
    return [[x, y] for _, x, y in heapq.nsmallest(k, heap)]

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val; self.next = next

# === REVERSE LINKED LIST — Iterative ===
def reverse_list(head):
    prev, curr = None, head
    while curr:
        nxt = curr.next
        curr.next = prev
        prev, curr = curr, nxt
    return prev

# === DETECT CYCLE — Floyd's Algorithm ===
def has_cycle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow == fast: return True
    return False

# === FIND MIDDLE — Fast/Slow Pointer ===
def find_middle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow

# === MERGE TWO SORTED LISTS ===
def merge_lists(l1, l2):
    dummy = ListNode(); cur = dummy
    while l1 and l2:
        if l1.val <= l2.val: cur.next = l1; l1 = l1.next
        else: cur.next = l2; l2 = l2.next
        cur = cur.next
    cur.next = l1 or l2
    return dummy.next

# === Nth FROM END — Two pointers ===
def remove_nth_from_end(head, n):
    dummy = ListNode(0, head)
    l, r = dummy, head
    for _ in range(n): r = r.next
    while r: l = l.next; r = r.next
    l.next = l.next.next
    return dummy.next

Linked List Interview Techniques

  • Dummy head node: simplifies edge cases when head might be modified (delete node, merge lists)
  • Fast/Slow pointers: find middle, detect cycle, find cycle start (Floyd's algorithm)
  • Reverse in groups: reverse k nodes at a time — iterate, reverse, connect three sections
  • Two-pass technique: first pass counts length, second pass finds position — O(n) with O(1) space
TOPIC 12

Trees

BST · AVL · SEGMENT TREE · TRIE · TRAVERSALS

TYPE

Binary Tree

Each node has 0–2 children. Height h = O(log n) for balanced, O(n) worst case. All BST/AVL trees are binary trees.

TYPE

Binary Search Tree

left < root < right for every node. O(log n) search/insert/delete on average. O(n) worst case (skewed). Self-balancing variants fix this.

TYPE

AVL / Red-Black Tree

Self-balancing BSTs. AVL: strict balance (height diff ≤ 1). Red-Black: relaxed rules, fewer rotations. Both guarantee O(log n) operations.

TYPE

Trie (Prefix Tree)

Each node represents a character. O(L) insert/search where L = word length. Used for autocomplete, spell check, IP routing, word games.

TYPE

Segment Tree

Range queries (sum, min, max) and point/range updates in O(log n). Build in O(n). Uses 4n nodes array. Essential for competitive programming.

TYPE

Fenwick Tree (BIT)

Prefix sums and updates in O(log n). More space-efficient than segment tree. Uses binary indexing trick. Limited to associative/invertible ops.

class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val; self.left = left; self.right = right

# === TREE TRAVERSALS ===
def inorder(root):    # Left → Root → Right (sorted for BST)
    return inorder(root.left) + [root.val] + inorder(root.right) if root else []

def level_order(root): # BFS level by level
    if not root: return []
    q, res = deque([root]), []
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft()
            level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res

# === MAX DEPTH / HEIGHT ===
def max_depth(root):
    if not root: return 0
    return 1 + max(max_depth(root.left), max_depth(root.right))

# === LOWEST COMMON ANCESTOR ===
def lca(root, p, q):
    if not root or root in (p, q): return root
    L = lca(root.left, p, q); R = lca(root.right, p, q)
    return root if L and R else L or R

# === TRIE IMPLEMENTATION ===
class Trie:
    def __init__(self):
        self.root = {}

    def insert(self, word):
        node = self.root
        for c in word:
            if c not in node: node[c] = {}
            node = node[c]
        node['#'] = True    # end of word marker

    def search(self, word):
        node = self.root
        for c in word:
            if c not in node: return False
            node = node[c]
        return '#' in node
